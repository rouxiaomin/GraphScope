28 class LDBCTimeStampParser : public arrow::TimestampParser {
29  public:
30  LDBCTimeStampParser() = default;
31 
32  ~LDBCTimeStampParser() override {}
33 
34  bool operator()(const char* s, size_t length, arrow::TimeUnit::type out_unit,
35  int64_t* out,
36  bool* out_zone_offset_present = NULLPTR) const override {
37  using seconds_type = std::chrono::duration<arrow::TimestampType::c_type>;
38 
39  // We allow the following formats for all units:
40  // - "YYYY-MM-DD"
41  // - "YYYY-MM-DD[ T]hhZ?"
42  // - "YYYY-MM-DD[ T]hh:mmZ?"
43  // - "YYYY-MM-DD[ T]hh:mm:ssZ?"
44  //
45  // We allow the following formats for unit == MILLI, MICRO, or NANO:
46  // - "YYYY-MM-DD[ T]hh:mm:ss.s{1,3}Z?"
47  //
48  // We allow the following formats for unit == MICRO, or NANO:
49  // - "YYYY-MM-DD[ T]hh:mm:ss.s{4,6}Z?"
50  //
51  // We allow the following formats for unit == NANO:
52  // - "YYYY-MM-DD[ T]hh:mm:ss.s{7,9}Z?"
53  //
54  // UTC is always assumed, and the DataType's timezone is ignored.
55  //
56 
57  if (ARROW_PREDICT_FALSE(length < 10))
58  return false;
59 
60  seconds_type seconds_since_epoch;
61  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseYYYY_MM_DD(
62  s, &seconds_since_epoch))) {
63  return false;
64  }
65 
66  if (length == 10) {
67  *out =
68  arrow::util::CastSecondsToUnit(out_unit, seconds_since_epoch.count());
69  return true;
70  }
71 
72  if (ARROW_PREDICT_FALSE(s[10] != ' ') &&
73  ARROW_PREDICT_FALSE(s[10] != 'T')) {
74  return false;
75  }
76 
77  if (s[length - 1] == 'Z') {
78  --length;
79  }
80 
81  // if the back the s is '+0000', we should ignore it
82  auto time_zones = std::string_view(s + length - 5, 5);
83  if (time_zones == "+0000") {
84  length -= 5;
85  }
86 
87  seconds_type seconds_since_midnight;
88  switch (length) {
89  case 13: // YYYY-MM-DD[ T]hh
90  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseHH(
91  s + 11, &seconds_since_midnight))) {
92  return false;
93  }
94  break;
95  case 16: // YYYY-MM-DD[ T]hh:mm
96  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseHH_MM(
97  s + 11, &seconds_since_midnight))) {
98  return false;
99  }
100  break;
101  case 19: // YYYY-MM-DD[ T]hh:mm:ss
102  case 21: // YYYY-MM-DD[ T]hh:mm:ss.s
103  case 22: // YYYY-MM-DD[ T]hh:mm:ss.ss
104  case 23: // YYYY-MM-DD[ T]hh:mm:ss.sss
105  case 24: // YYYY-MM-DD[ T]hh:mm:ss.ssss
106  case 25: // YYYY-MM-DD[ T]hh:mm:ss.sssss
107  case 26: // YYYY-MM-DD[ T]hh:mm:ss.ssssss
108  case 27: // YYYY-MM-DD[ T]hh:mm:ss.sssssss
109  case 28: // YYYY-MM-DD[ T]hh:mm:ss.ssssssss
110  case 29: // YYYY-MM-DD[ T]hh:mm:ss.sssssssss
111  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseHH_MM_SS(
112  s + 11, &seconds_since_midnight))) {
113  return false;
114  }
115  break;
116  default:
117  LOG(ERROR) << "unsupported length: " << length;
118  return false;
119  }
120 
121  seconds_since_epoch += seconds_since_midnight;
122 
123  if (length <= 19) {
124  *out =
125  arrow::util::CastSecondsToUnit(out_unit, seconds_since_epoch.count());
126  return true;
127  }
128 
129  if (ARROW_PREDICT_FALSE(s[19] != '.')) {
130  return false;
131  }
132 
133  uint32_t subseconds = 0;
134  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseSubSeconds(
135  s + 20, length - 20, out_unit, &subseconds))) {
136  return false;
137  }
138 
139  *out =
140  arrow::util::CastSecondsToUnit(out_unit, seconds_since_epoch.count()) +
141  subseconds;
142  return true;
143  }
144 
145  const char* kind() const override { return "LDBC timestamp parser"; }
146 
147  const char* format() const override { return "EmptyFormat"; }
148 };

150 class LDBCLongDateParser : public arrow::TimestampParser {
151  public:
152  using seconds_type = std::chrono::duration<arrow::TimestampType::c_type>;
153  LDBCLongDateParser() = default;
154 
155  ~LDBCLongDateParser() override {}
156 
157  bool operator()(const char* s, size_t length, arrow::TimeUnit::type out_unit,
158  int64_t* out,
159  bool* out_zone_offset_present = NULLPTR) const override {
160  uint64_t seconds;
161  // convert (s, s + length - 4) to seconds_since_epoch
162  if (ARROW_PREDICT_FALSE(
163  !arrow::internal::ParseUnsigned(s, length - 3, &seconds))) {
164  return false;
165  }
166 
167  uint32_t subseconds = 0;
168  if (ARROW_PREDICT_FALSE(!arrow::internal::detail::ParseSubSeconds(
169  s + length - 3, 3, out_unit, &subseconds))) {
170  return false;
171  }
172 
173  *out = arrow::util::CastSecondsToUnit(out_unit, seconds) + subseconds;
174  return true;
175  }
176 
177  const char* kind() const override { return "LDBC timestamp parser"; }
178 
179  const char* format() const override { return "LongDateFormat"; }
180 };
