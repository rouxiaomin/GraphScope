174  Allocator& alloc) override {
175  auto oe = out_csr_->edge_iter_mut(src);
176  EDATA_T prop;

178  bool src_flag = false, dst_flag = false;
179  while (oe != nullptr && oe->is_valid()) {
180  if (oe->get_neighbor() == dst) {
181  oe->set_data(prop, ts);
182  src_flag = true;
183  break;
184  }
185  oe->next();
186  }
187  auto ie = in_csr_->edge_iter_mut(dst);
188  while (ie != nullptr && ie->is_valid()) {
189  if (ie->get_neighbor() == src) {
190  dst_flag = true;
191  ie->set_data(prop, ts);
192  break;
193  }
194  ie->next();
195  }
196  if (!(src_flag || dst_flag)) {
197  in_csr_->put_edge(dst, src, prop, ts, alloc);
198  out_csr_->put_edge(src, dst, prop, ts, alloc);
199  }
