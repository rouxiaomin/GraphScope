212  void from_timestamp(int64_t ts) {
213  const boost::posix_time::ptime epoch(boost::gregorian::date(1970, 1, 1));
214  int64_t ts_sec = ts / 1000;
215  boost::posix_time::ptime time_point =
216  epoch + boost::posix_time::seconds(ts_sec);
217  boost::posix_time::ptime::date_type date = time_point.date();
218  boost::posix_time::time_duration td = time_point.time_of_day();
219  this->value.internal.year = date.year();
220  this->value.internal.month = date.month().as_number();
221  this->value.internal.day = date.day();
222  this->value.internal.hour = td.hours();
223 
224  int64_t ts_back = to_timestamp();
225  CHECK_EQ(ts, ts_back);
226  }
